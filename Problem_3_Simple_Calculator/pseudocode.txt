num1=int(input("enter the first number:"))
num2=int(input("enter the second number:"))
operator=input("enter the operator(+,-,*,/):")
if operator =='+':
    print(num1 + num2)
elif operator =='-':
    print(num1 - num2)
elif operator =='*':
    print(num1 * num2)
elif operator =='/':
    if num2 != 0:
        print(num1 / num2)
    else:
        print("error! division by zero")
else:
    print("inavalid operator")
